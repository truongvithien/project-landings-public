Integration Example

import { LoadingScreenGSAP, useLoadingScreenGSAP } from '@/components/ui/loading-screen-gsap';
import { useAssetPreloader, AssetConfig } from '@/lib/utils/asset-preloader';

const MyComponent = () => {
  const { startLoading, updateProgress, finishLoading } = useLoadingScreenGSAP();
  const { preloadAssetsWithProgress } = useAssetPreloader();

  const assets: AssetConfig[] = [
    { url: '/hero-image.jpg', type: 'image', priority: 'high' },
    { url: '/video.mp4', type: 'video', priority: 'medium' },
    { url: '/model.glb', type: 'model', priority: 'low' },
  ];

  const handlePreload = async () => {
    startLoading({ initialMessage: 'Loading assets...' });
    
    try {
      const results = await preloadAssetsWithProgress({ assets });
      finishLoading();
    } catch (error) {
              // console.error('Preload failed:', error);
      finishLoading();
    }
  };

  return (
    <div>
      <button onClick={handlePreload}>Preload Assets</button>
      {/* Loading screen will show automatically */}
    </div>
  );
};
