Usage Instructions

Basic Loading Screen

import { LoadingScreen, useLoadingScreen } from '@/components/ui/loading-screen';

const { isLoading, progress, message, startLoading, updateProgress, finishLoading } = useLoadingScreen();

// Start loading
startLoading({ initialMessage: 'Loading assets...' });

// Update progress
updateProgress({ progress: 50, message: 'Loading textures...' });

// Finish loading
finishLoading();

GSAP Enhanced Loading Screen

import { LoadingScreenGSAP, useLoadingScreenGSAP } from '@/components/ui/loading-screen-gsap';

const { isLoading, progress, message, startLoading, updateProgress, finishLoading, handleComplete } = useLoadingScreenGSAP();

// Start loading
startLoading({ initialMessage: 'Loading assets...' });

// Update progress
updateProgress({ progress: 50, message: 'Loading textures...' });

// Finish loading (GSAP will handle the completion animation)
finishLoading();
